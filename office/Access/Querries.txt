-- Ερώτημα 1: Εμφάνιση όλων των στοιχείων ενός πελάτη
SELECT * FROM Πελάτες
WHERE ΚΩΔ_ΠΕΛ = [Δώστε κωδικό πελάτη:];

-- Ερώτημα 2: Εμφάνιση όλων των στοιχείων των πελατών το τηλέφωνο των οποίων αρχίζει από ένα πρόθεμα
SELECT * FROM Πελάτες
WHERE ΤΗΛΕΦΩΝΟ LIKE [Τηλέφωνο πελάτη αρχίζει από:] + '%';

-- Ερώτημα 3: Εμφάνιση όλων των στοιχείων των πελατών το τηλέφωνο των οποίων περιλαμβάνει ένα νούμερο
SELECT * FROM Πελάτες
WHERE ΤΗΛΕΦΩΝΟ LIKE '%' + [Τηλέφωνο περιέχει:] + '%';

-- Ερώτημα 4: Εμφάνιση όλων των προϊόντων με τιμή μικρότερη από μία τιμή
SELECT * FROM Προϊόντα
WHERE ΤΙΜΗ_ΜΟΝΑΔΑΣ < [Προϊόντα με τιμή μικρότερη από:];

-- Ερώτημα 5: Εμφάνιση όλων των προϊόντων με τιμή μικρότερη από μία τιμή και μεγαλύτερη από μία άλλη
SELECT * FROM Προϊόντα
WHERE ΤΙΜΗ_ΜΟΝΑΔΑΣ < [Προϊόντα με τιμή μικρότερη από:]
AND ΤΙΜΗ_ΜΟΝΑΔΑΣ > [Προϊόντα με τιμή μεγαλύτερη από:];

-- Ερώτημα 6: Εμφάνιση όλων των παραγγελιών με ημερομηνία αγοράς μεταξύ δύο ημερομηνιών
SELECT * FROM Παραγγελίες
WHERE ΗΜΕΡ_ΑΓΟΡΑΣ BETWEEN [Παραγγελία από:] AND [Παραγγελία έως:];
