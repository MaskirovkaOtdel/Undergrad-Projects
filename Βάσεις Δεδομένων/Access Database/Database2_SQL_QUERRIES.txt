ΕΡΩΤΗΜΑ_1
SELECT MAX([ΤΙΜΗ_ΜΟΝΑΔΑΣ]) AS [Μέγιστη_Τιμή]
FROM ΠΑΡΑΓΓΕΛΙΑ;


ΕΡΩΤΗΜΑ_2
SELECT MIN([ΤΙΜΗ_ΜΟΝΑΔΑΣ]) AS [Ελάχιστη_Τιμή]
FROM ΠΑΡΑΓΓΕΛΙΑ;


ΕΡΩΤΗΜΑ_3
SELECT AVG([ΤΙΜΗ_ΜΟΝΑΔΑΣ]) AS [Μέση_Τιμή]
FROM ΠΑΡΑΓΓΕΛΙΑ;


ΕΡΩΤΗΜΑ_4
SELECT COUNT(*) AS [Συνολικές_Παραγγελίες]
FROM ΠΑΡΑΓΓΕΛΙΑ;


ΕΡΩΤΗΜΑ_5
SELECT [ΗΜΕΡ_ΑΓΟΡΑΣ], COUNT(*) AS [Παραγγελίες]
FROM ΠΑΡΑΓΓΕΛΙΑ
GROUP BY [ΗΜΕΡ_ΑΓΟΡΑΣ];


 ΕΡΩΤΗΜΑ_6
SELECT COUNT(*) AS [Παραγγελίες_18102004]
FROM ΠΑΡΑΓΓΕΛΙΑ
WHERE [ΗΜΕΡ_ΑΓΟΡΑΣ] = #18/10/2004#;


 ΕΡΩΤΗΜΑ_7
SELECT ΠΑΡΑΓΓΕΛΙΑ.*, 
       [ΠΟΣΟΤΗΤΑ_ΠΡΟΙΟΝΤΟΣ] * [ΤΙΜΗ_ΜΟΝΑΔΑΣ] AS [ΣΥΝΟΛΙΚΟ_ΚΟΣΤΟΣ]
FROM ΠΑΡΑΓΓΕΛΙΑ;


ΕΡΩΤΗΜΑ_8
SELECT ΠΑΡΑΓΓΕΛΙΑ.*, 
       [ΠΟΣΟΤΗΤΑ_ΠΡΟΙΟΝΤΟΣ] * [ΤΙΜΗ_ΜΟΝΑΔΑΣ] AS [ΣΥΝΟΛΙΚΟ_ΚΟΣΤΟΣ],
       ROUND(([ΠΟΣΟΤΗΤΑ_ΠΡΟΙΟΝΤΟΣ] * [ΤΙΜΗ_ΜΟΝΑΔΑΣ]) - 
             ([ΠΟΣΟΤΗΤΑ_ΠΡΟΙΟΝΤΟΣ] * [ΤΙΜΗ_ΜΟΝΑΔΑΣ]) / 1.19, 3) AS [ΣΥΝΟΛΙΚΟ_ΦΠΑ]
FROM ΠΑΡΑΓΓΕΛΙΑ;


 ΕΡΩΤΗΜΑ_9
SELECT SUM([ΠΟΣΟΤΗΤΑ_ΠΡΟΙΟΝΤΟΣ]) AS [Συνολικά_Τετράδια]
FROM ΠΑΡΑΓΓΕΛΙΑ
WHERE [ΠΡΟΙΟΝ] = 'ΤΕΤΡΑΔΙΟ';


 ΕΡΩΤΗΜΑ_10
SELECT ΠΑΡΑΓΓΕΛΙΑ.*, 
       [ΕΠΩΝΥΜΟ] & ' ' & [ΟΝΟΜΑ] AS [ΟΝΟΜΑΤΕΠΩΝΥΜΟ]
FROM ΠΑΡΑΓΓΕΛΙΑ;
