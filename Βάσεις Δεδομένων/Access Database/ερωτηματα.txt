SELECT Max([ΤΙΜΗ_ΜΟΝΑΔΑΣ]) AS Μέγιστη_Τιμή
FROM ΠΑΡΑΓΓΕΛΙΑ;

SELECT Min([ΤΙΜΗ_ΜΟΝΑΔΑΣ]) AS Ελάγχιστη_Τιμή
FROM ΠΑΡΑΓΓΕΛΙΑ;


SELECT Avg([ΤΙΜΗ_ΜΟΝΑΔΑΣ]) AS Μέση_Τιμή
FROM ΠΑΡΑΓΓΕΛΙΑ;


SELECT Count(*) AS Συνολικές_Παραγγελίες
FROM ΠΑΡΑΓΓΕΛΙΑ;

SELECT [ΗΜΕΡ_ΑΓΟΡΑΣ], Count(*) AS Παραγγελίες
FROM ΠΑΡΑΓΓΕΛΙΑ
GROUP BY [ΗΜΕΡ_ΑΓΟΡΑΣ];

SELECT Count(*) AS Παραγγελίες_18102004
FROM ΠΑΡΑΓΓΕΛΙΑ
WHERE [ΗΜΕΡ_ΑΓΟΡΑΣ] = #10/18/2004#;


SELECT ΠΑΡΑΓΓΕΛΙΑ.*, [ΠΟΣΟΤΗΤΑ_ΠΡΟΪΟΝΤΟΣ] * [ΤΙΜΗ_ΜΟΝΑΔΑΣ] AS ΣΥΝΟΛΙΚΟ_ΚΟΣΤΟΣ
FROM ΠΑΡΑΓΓΕΛΙΑ;

SELECT ΠΑΡΑΓΓΕΛΙΑ.*, [ΠΟΣΟΤΗΤΑ_ΠΡΟΪΟΝΤΟΣ] * [ΤΙΜΗ_ΜΟΝΑΔΑΣ] AS ΣΥΝΟΛΙΚΟ_ΚΟΣΤΟΣ, ROUND(([ΠΟΣΟΤΗΤΑ_ΠΡΟΪΟΝΤΟΣ] * [ΤΙΜΗ_ΜΟΝΑΔΑΣ]) - ([ΠΟΣΟΤΗΤΑ_ΠΡΟΪΟΝΤΟΣ] * [ΤΙΜΗ_ΜΟΝΑΔΑΣ]) / 1.19, 3) AS ΣΥΝΟΛΙΚΟ_ΦΠΑ
FROM ΠΑΡΑΓΓΕΛΙΑ;


SELECT SUM([ΠΟΣΟΤΗΤΑ_ΠΡΟΙΟΝΤΟΣ]) AS Συνολικά_Τετράδια
FROM ΠΑΡΑΓΓΕΛΙΑ
WHERE [ΠΡΟΙΟΝ] = 'ΤΕΤΡΑΔΙΟ';


SELECT ΠΑΡΑΓΓΕΛΙΑ.*, [ΕΠΩΝΥΜΟ] & ' ' & [ΟΝΟΜΑ] AS ΟΝΟΜΑΤΕΠΩΝΥΜΟ
FROM ΠΑΡΑΓΓΕΛΙΑ;
