SELECT "ΘΥΡΕΣ", COUNT( * ) AS "ΣΥΝΟΛΟ_ΑΥΤΟΚΙΝΗΤΩΝ" FROM "ΑΥΤΟΚΙΝΗΤΑ" WHERE "ΘΥΡΕΣ" IN ( 3, 5 ) GROUP BY "ΘΥΡΕΣ"

SELECT "ΜΑΡΚΑ", AVG( "ΚΥΒΙΚΑ" ) AS "Μεσος_Ορος_Κυβικων" FROM "ΑΥΤΟΚΙΝΗΤΑ" GROUP BY "ΜΑΡΚΑ"

SELECT "ΜΑΡΚΑ", COUNT( * ) AS "Συνολο_Αυτοκινητων" FROM "ΑΥΤΟΚΙΝΗΤΑ" WHERE "ΚΥΒΙΚΑ" BETWEEN 1000 AND 2000 GROUP BY "ΜΑΡΚΑ"

SELECT "ΧΡΩΜΑ", SUM( "ΚΥΒΙΚΑ" ) AS "Συνολικά_Κυβικά" FROM "ΑΥΤΟΚΙΝΗΤΑ" GROUP BY "ΧΡΩΜΑ"

SELECT "ΜΑΡΚΑ", COUNT( * ) AS "Συνολο_Αυτοκινητων" FROM "ΑΥΤΟΚΙΝΗΤΑ" GROUP BY "ΜΑΡΚΑ"

SELECT "ΜΑΡΚΑ", "ΜΟΝΤΕΛΟ", COUNT( * ) AS "Αριθμός_Μοντέλων" FROM "ΑΥΤΟΚΙΝΗΤΑ" WHERE "ΜΟΝΤΕΛΟ" LIKE '%X%' GROUP BY "ΜΑΡΚΑ", "ΜΟΝΤΕΛΟ"

SELECT "ΜΑΡΚΑ", MIN( "ΕΤΟΣ_ΚΥΚΛ" ) AS "Παλαιότερο", MAX( "ΕΤΟΣ_ΚΥΚΛ" ) AS "Νεότερο" FROM "ΑΥΤΟΚΙΝΗΤΑ" GROUP BY "ΜΑΡΚΑ"
